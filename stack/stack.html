public class Stack {
    private Object[] stack;
    private int size;
    private int capacity;
    private int top;

    public Stack() {
        this.capacity = 5;
        this.stack = new Object[this.capacity];
        this.size = 0;
        this.top = -1;
    }

    private void resize() {
        int newCapacity = capacity * 2;
        this.stack = Arrays.copyOf(this.stack, newCapacity);
        this.capacity = newCapacity;
    }

    public void push(T data) {
        if (size == capacity) {
            resize();
        }
        top++;
        stack[top] = data;
        size++;
    }

    public T pop() {
        if (isEmpty()) {
            return null;
        }
        T data = (T) stack[top];
        stack[top] = null;
        top--;
        size--;
        return data;
    }

    public T peek() {
        if (isEmpty()) {
            return null;
        }
        return (T) stack[top];
    }

    public boolean isEmpty() {
        return size == 0;
    }
    
    public int size() {
        return size;
    }
}
